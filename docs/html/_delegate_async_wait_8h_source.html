281 virtual RetType operator()(Args... args) override {
282 if (this->Empty())
283 return RetType();
284
285 // Synchronously invoke the target function?
286 if (this->GetSync()) {
287 // Invoke the target function directly
288 return BaseType::operator()(std::forward<Args>(args)...);
289 } else {
290 // Create a clone instance of this delegate
291 auto delegate = std::shared_ptr<ClassType>(Clone());
292 if (!delegate)
293 throw std::bad_alloc();
294
295 // Create a new message instance for sending to the destination thread.
296 auto msg = std::make_shared<DelegateAsyncWaitMsg<Args...>>(delegate, std::forward<Args>(args)...);
297 if (!msg)
298 throw std::bad_alloc();
299 msg->SetInvokerWaiting(true);
300
301 auto thread = this->GetThread();
302 if (thread) {
303 // Dispatch message onto the callback destination thread. Invoke()
304 // will be called by the destination thread.
305 thread->DispatchDelegate(msg);
306
307 // Wait for destination thread to execute the delegate function and get return value
308 if ((m_success = msg->GetSema().Wait(m_timeout)))
309 m_retVal = delegate->m_retVal;
310 }
311
312 // Protect data shared between source and destination threads
313 const std::lock_guard<std::mutex> lock(msg->GetLock());
314
315 // Set flag that source is not waiting anymore
316 msg->SetInvokerWaiting(false);
317
318 // Does the target function have a return value?
319 if constexpr (std::is_void<RetType>::value == false) {
320 // Is the return value valid?
321 if (m_retVal.has_value()) {
322 // Return the destination thread target function return value
323 return GetRetVal();
324 } else {
325 // Return a default return value
326 return RetType{};
327 }
328 }
329 }

359 virtual bool Invoke(std::shared_ptr<DelegateMsg> msg) override {
360 static_assert(!(is_unique_ptr<RetType>::value), "std::unique_ptr return value not allowed");
361
362 // Typecast the base pointer to back correct derived to instance
363 auto delegateMsg = std::dynamic_pointer_cast<DelegateAsyncWaitMsg<Args...>>(msg);
364 if (delegateMsg == nullptr)
365 return false;
366
367 // Protect data shared between source and destination threads
368 const std::lock_guard<std::mutex> lock(delegateMsg->GetLock());
369
370 // Is the source thread waiting for the target function invoke to complete?
371 if (delegateMsg->GetInvokerWaiting()) {
372 // Invoke the delegate function synchronously
373 this->SetSync(true);
374
375 // Does target function have a void return value?
376 if constexpr (std::is_void<RetType>::value == true) {
377 // Invoke the target function using the source thread supplied function arguments
378 std::apply(&BaseType::operator(), std::tuple_cat(std::make_tuple(this), delegateMsg->GetArgs()));

383 }

389 }
